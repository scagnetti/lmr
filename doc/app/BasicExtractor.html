def open_sub_page(link_text, result_size, link_index)
  links = @stock_page.links_with(:text => link_text)
  if links == nil || links.size() != result_size
    raise DataMiningError, "Could not find link containing text >>#{link_text}<<", caller
  else
    sub_page = links[link_index].click()
    return sub_page
  end
end

def perform_search(isin, attribute, regexp, field_name, failure_msg)
  search_form = @start_page.form_with(attribute => regexp)
  search_form[field_name] = isin
  result = search_form.submit
  failure = result.parser().xpath(failure_msg)
  if failure != nil && failure.size() > 0
    raise InvalidIsin, "Could not find stock page with supplied ISIN: #{isin}", caller
  else
    return result
  end 
end
