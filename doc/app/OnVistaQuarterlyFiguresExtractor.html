def initialize(agent, score_card, on_vista_stock_id, on_vista_index_id)

def go()
  reaction = Reaction.new
  
  raw_dates = get_release_dates()
  
  casted_dates = convert_to_time(raw_dates)
  
  date = get_latest(casted_dates, THREE_MONTHS_IN_SECONDS)
  
  if date == Time.at(0)
    # No release date in the 3 month interval, this is a serious problem
    raise DataMiningError, "Could not find quarterly figures within 3 months", caller
  else
    reaction.release_date = date  
  end

  # Get the value of the stock when quarterly figures where published
  stock_opening_closing = @value_extractor.extract_stock_value_on(date)
  reaction.price_opening = stock_opening_closing[0]
  reaction.price_closing = stock_opening_closing[1]
  
  # Get the value of the index when quaterly figures where published
  index_opening_closing = @value_extractor.extract_index_value_on(date)
  reaction.index_opening = index_opening_closing[0]
  reaction.index_closing = index_opening_closing[1]

  return reaction
end
