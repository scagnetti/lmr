def initialize(agent, on_vista_id, on_vista_index_id)

def extract_index_value_on(historical_date)
  historical_date = ensure_string(historical_date)
  # Enter search date
  history_page = @index_search_page.form_with(:action => %r^historie\.html\?ID_NOTATION=.*$/) do |f|
    f.DATE = historical_date
  end.click_button
  
  d = adjust_supplied_date(historical_date)
  tag_set = history_page.parser().xpath("//td[starts-with(.,'#{d}')]/following-sibling::td")
  if tag_set == nil || tag_set.size() != 10
    raise DataMiningError, "Could not get a index value for the given date (#{d})", caller
  end
  LOG.debug("Index opening value #{tag_set[1].content()}")
  LOG.debug("Index closing value #{tag_set[7].content()}")
  opening = Util.l10n_f_k(tag_set[1].content().strip())
  closing = Util.l10n_f_k(tag_set[7].content().strip())
  return [opening, closing]
end

def extract_stock_value_on(historical_date)
  historical_date = ensure_string(historical_date)
  # Enter search date
  history_page = @stock_search_page.form_with(:action => %r.*kurshistorie$/) do |f|
    f.DATE  = historical_date
  end.click_button
  
  d = adjust_supplied_date(historical_date)
  tag_set = history_page.parser().xpath("//td[. = '#{d}']/following-sibling::td")
  if tag_set == nil || tag_set.size() != 4
    raise DataMiningError, "Could not get a stock value for the given date (#{d})", caller
  end
  LOG.debug("Opening value #{tag_set[0].content()}")
  LOG.debug("Closing value #{tag_set[3].content()}")
  opening = Util.l10n_f(tag_set[0].content().strip())
  closing = Util.l10n_f(tag_set[3].content().strip())
  return [opening, closing]
end
