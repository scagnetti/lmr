def extract_reaction_on_figures(index_isin)
  index_page = perform_search(index_isin, "action", '/suche/', "SEARCH_VALUE", SEARCH_FAILURE)
  @on_vista_index_id = get_index_id(index_page)
  @on_vista_id = get_stock_id(@stock_page)
  figuresExtractor = OnVistaQuarterlyFiguresExtractor.new(@agent, @score_card, @on_vista_id, @on_vista_index_id)
  return figuresExtractor.go()
end

def extract_roe()
  key_figures_page = open_sub_page('Kennzahlen', 2, 1)
  roe = 1000
  tag_set = key_figures_page.parser().xpath("//td[.='Eigenkapitalrendite in %']/following-sibling::td")
  if tag_set == nil || tag_set.size() != 3
    raise DataMiningError, "Could not extract RoE", caller
  else
    raw_roe = tag_set[0].content().strip()
    raise DataMiningError, "Could not extract RoE", caller if raw_roe == "n.a."
    roe = Util.l10n_f(raw_roe)
    LOG.debug("RoE LJ: #{roe}")
  end
  return_on_equity = ReturnOnEquity.new
  return_on_equity.value = roe
end

def extract_stock_name()
  scan_result = @stock_page.title().split(%r-/)
  if scan_result.size > 0
    return scan_result[0]
  else
    raise DataMiningError, "Search result was not a stock page!", caller 
  end
end

def extract_stock_price()
  price_now = -1
  tag_set = @stock_page.parser().xpath("//span[@class='KURSRICHTUNG']/following-sibling::span")
  if tag_set == nil || tag_set.size() != 1
    raise DataMiningError, "Could no extract current stock price", caller
  else
    raw_price_now = tag_set[0].content()
    price_now = Util.l10n_f(raw_price_now)
  end
  return price_now
end
